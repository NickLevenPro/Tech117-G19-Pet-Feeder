Introduction

Owning cats can be a very time-consuming endeavour. Nevermind footing the bill for their healthcare, vaccines, toys, and bathroom needs, just feeding them can be a chore. And God-forbid you forget to feed them, your ears will experience relentless harassment as the force of a thousand meows penetrates your eardrums. Wouldn’t it be nice to have a cat feeder to prevent this situation? Taking it a step further, wouldn’t it be nice to be able to assemble this device yourself using common, cheap, and recyclable materials? This cat feeder does all that, ensuring your ferocious feline’s dietary needs are met while maximizing sustainability.

Prototype

Code

  
  
#include <LiquidCrystal.h>
#include <Servo.h>
#define buzzer 10
Servo myservo;
int pos = 30;  
int seconds = 0;

// Delay configuration variable (in seconds)
int feedDelay = 60;  // Set delay time here (in seconds)
unsigned long delayStartTime = 0;
boolean delayActive = true;

LiquidCrystal lcd_1(12, 11, 5, 4, 3, 2);

void setup()
{
  myservo.attach(9);
  lcd_1.begin(16, 2);
  myservo.write(pos);
}

void loop()
{
  // Check if delay countdown is active
  if (delayActive) {
    displayCountdown();
  } else {
    // Display ready message
    lcd_1.setCursor(0, 0);
    lcd_1.print("Ready to Feed!");
    lcd_1.setCursor(0, 1);
    lcd_1.print("                ");
    delay(5000);
    delayStartTime = millis();
    delayActive = true;
  }
}

// Function to display countdown on LCD
void displayCountdown() {
  unsigned long elapsedTime = (millis() - delayStartTime) / 1000;
  int remainingTime = feedDelay - elapsedTime;
 
  if (remainingTime > 0) {
    lcd_1.setCursor(0, 0);
    lcd_1.print("Feeding in:     ");
    lcd_1.setCursor(0, 1);
    lcd_1.print(remainingTime);
    lcd_1.print(" sec    ");
    delay(100); // Update display every 100ms
  } else {
    // Countdown finished - buzz and activate feeding
    delayActive = false;
    //buzzAlert();
    feedPets();
  }
}

// Function to buzz when countdown is done
void buzzAlert() {
  tone(buzzer, 500, 500);  // 500Hz for 500ms
  delay(100);
  tone(buzzer, 500, 500);  // Double buzz
  delay(100);
}

// Function to activate servo and feed
void feedPets() {
  // rotate slowly to dispense
  for (pos = 30; pos <= 80; pos += 1) {
    myservo.write(pos);
    delay(15);
  }
  delay(100); // short pause at max

  // return quickly
  for (pos = 80; pos >= 30; pos -= 10) {
    myservo.write(pos);
    delay(20);
  }
  // ensure fully returned
  myservo.write(30);
 

  delayStartTime = millis();
  delayActive = false; 
  
  
}

Conclusion

The smart pet feeder successfully provides an easy and reliable way to feed pets automatically. It helps pet owners save time and ensures pets are fed on a regular schedule. Overall, this project demonstrates how simple technology can be used to solve real life problems in a practical and effective way.
